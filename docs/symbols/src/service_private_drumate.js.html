 71 // change_email
 72 // Changes existing email of a drumate.
 73 // ========================
 74   change_email() {
 75     const email         = this.input.need(_a.email);
 76     
 77     if (!email.isEmail()) {
 78       this.exception.user(_k.INVALID_EMAIL_FORMAT);
 79       return;
 80     }
 81     this.yp.call_proc('email_exists', email, function(row){ 
 82       if (!_.isEmpty(row) && row.email) {
 83         this.exception.user(_k.EMAIL_ALREADY_EXIST);
 84       } else {
 85         this.yp.call_proc('drumate_change_email', this.uid, email,  this.output.data);
 86       }
 87     }.bind(this));
 88   }

116     if (!new_password.match(/(.+){8,}/)) { //(/(.+){2,} +(.+){4,}/)

118         code : 412,

122     } else {
123       this.yp.call_proc('set_password', this.uid, new_password, this.output.data);

373   /**
374    * Set user avatar using a media available in MFS
375    * @param {string} reference - media id from MFS
376    */
377   async set_avatar() {
378     const nid = this.input.use('reference');
379     let node = await this.db.await_proc("mfs_node_attr", nid);
380     if(_.isEmpty(node)){
381       this.exception.not_found("no_avatar");
382       return;
383     }
384     const src = Vfs.get_node_content(node);
385     if(!Fs.existsSync(src)){
386       this.exception.not_found("no_avatar");
387       return;
388     }
389     const root = node.home_dir.replace('/__storage__/', '');
390     let dest;
391     if(/svg/i.test(node.ext)){
392       dest = Vfs.normalize_path(root + '/__config__/icons/avatar.svg', '');
393     }else{
394       dest = Vfs.normalize_path(root + '/__config__/icons/avatar.png', '');
395     }
396     const cmd = `gm convert -auto-orient -thumbnail '100x100^' -gravity center ${src} -extent 100x100 +profile \"*\" ${dest}`;
397     Vfs.exec(cmd);
398     await this.yp.await_proc('entity_touch', this.uid);
399     let data = await  this.yp.await_proc('get_visitor', this.uid);
400     this.output.data(data);
401   }
